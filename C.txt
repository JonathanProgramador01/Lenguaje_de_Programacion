#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#define SIZE 3
int main() {
	//O(n)

	int base[SIZE] = {0};

	for (int a = 0; a < SIZE; a++) {
		int guardar;
		scanf("%d", &guardar);
		base[a] = guardar;

	}

	int mayor = base[0];
	for (int a = 1; a < SIZE; a++) {
		if (mayor < base[a])
			mayor = base[a];

	}

	printf("TU NUMERO MAYOR ES: %d", mayor);


	return 0;
}
